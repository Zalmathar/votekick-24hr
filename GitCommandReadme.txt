git status //shows all files that have been changed or added and is ready to be commited
git add . //stages files in git status to be commited
git commit -m 'Insert comment here' //commits files that have been staged in the status command
git clone 'repository' //clones the repository from remote repository (GitHub)
git fetch origin //pulls any changes made in the repository, will not automaticlly murge.
git pull origin //pulls any changes made in repository and automaticlly local and repository files.
git push origin master //pushes any changes made localy to the master branch. Will ask for credentials
